\chapter{Multilinear KZG Polynomial Commitment}
\label{chap:mkzg}

Multilinear KZG (MKZG) extends KZG to polynomials over $\mu$
variables where each variable has degree at most one. Such
polynomials arise naturally in sumcheck-based proof systems and are
the foundation of HyperPlonk \cite{hyperplonk2022}. The SRS encodes
the multilinear Lagrange basis at a secret point rather than powers
of a single secret. The proof consists of $\mu$ group elements and
verification requires $\mu + 1$ pairings, both growing with the
number of variables but remaining much cheaper than the prover cost.

\section{Multilinear KZG Construction}

\textbf{Multilinear Polynomials:}

A multilinear polynomial in $\mu$ variables is uniquely determined
by its $2^\mu$ evaluations on the Boolean hypercube $\{0,1\}^\mu$.
Its multilinear extension is:
\begin{equation}
    \tilde{f}(\mathbf{X})
    = \sum_{\mathbf{b} \in \{0,1\}^\mu}
      \hat{f}(\mathbf{b}) \cdot \chi_{\mathbf{b}}(\mathbf{X}),
    \qquad
    \chi_{\mathbf{b}}(\mathbf{X})
    = \prod_{i=1}^{\mu}
      \bigl( X_i b_i + (1 - X_i)(1 - b_i) \bigr)
\end{equation}
\\
\textbf{Trusted Setup:}

The SRS is built from a secret vector $\mathbf{r} \in \mathbb{F}_r^\mu$:
\begin{itemize}
    \item $\mathsf{srs\_g1}[\mathbf{b}] = \chi_{\mathbf{b}}(\mathbf{r}) \cdot g_1$
          for each $\mathbf{b} \in \{0,1\}^\mu$ \quad ($2^\mu$ elements)
    \item $\mathsf{g2\_r}[i] = r_i \cdot g_2$ for $i = 0,\ldots,\mu-1$
          \quad ($\mu$ elements)
\end{itemize}
\\
\textbf{Commitment:}
The commitment is an MSM of size $2^\mu$:
\begin{equation}
    C = \sum_{\mathbf{b}} \hat{f}(\mathbf{b}) \cdot \chi_{\mathbf{b}}(\mathbf{r}) \cdot g_1
      = [\tilde{f}(\mathbf{r})]_1
\end{equation}
\\
\textbf{Proof:}
To prove $\tilde{f}(\mathbf{z}) = v$, the prover uses the identity:
\begin{equation}
    \tilde{f}(\mathbf{X}) - v
    = \sum_{i=0}^{\mu-1} (X_i - z_i) \cdot q_i(X_{i+1}, \ldots, X_{\mu-1})
\end{equation}
and commits to each quotient: $\pi_i = [\tilde{q}_i(\mathbf{r})]_1$.
\\
\textbf{Verification:}
The verifier checks (in additive $\mathbb{G}_T$ notation):
\begin{equation}
    e(C - v \cdot g_1,\; g_2)
    \stackrel{?}{=}
    \sum_{i=0}^{\mu-1} e\!\left(\pi_i,\; \mathsf{g2\_r}[i] - z_i \cdot g_2\right)
\end{equation}

\subsection{Bookkeeping Table for Witness Computation}

Rather than computing quotient polynomials explicitly (which would
require knowing $\mathbf{r}$), the implementation uses a bookkeeping
table. The table starts as $\hat{f}$ and is halved at each step:
\begin{equation}
    \mathsf{table}_{\mathsf{next}}[j]
    = (1 - z_i) \cdot \mathsf{table}[2j] + z_i \cdot \mathsf{table}[2j+1]
\end{equation}
At step $i$, the MSM weight for index $\mathsf{idx}$ is
$w[\mathsf{idx}] = \mathsf{table}[2j+1] - \mathsf{table}[2j]$
where $j$ encodes the upper bits of $\mathsf{idx}$. This avoids
any knowledge of $\mathbf{r}$ during proving.

\section{Pseudocode}

\begin{algorithm}[H]
\caption{MKZG Setup, Commit, Open, Verify}
\label{alg:mkzg-all}
\begin{algorithmic}[1]
\Statex \textbf{Setup}$(g_1, g_2, \mathbf{r}, \mu)$:
\For{$i = 0$ \textbf{to} $2^\mu - 1$}
    \State Decode $i$ to bits $\mathbf{b}$;\;
           $\mathsf{srs\_g1}.\mathsf{push}(\chi_{\mathbf{b}}(\mathbf{r}) \cdot g_1)$
\EndFor
\For{$i = 0$ \textbf{to} $\mu-1$}
    \State $\mathsf{g2\_r}.\mathsf{push}(r_i \cdot g_2)$
\EndFor;\; Discard $\mathbf{r}$
\Statex
\Statex \textbf{Commit}$(\mathsf{srs\_g1}, \hat{f})$:
\Return $\mathsf{MSM}(\mathsf{srs\_g1}.\mathsf{to\_affine}(),\; \hat{f})$
\Statex
\Statex \textbf{Open}$(\mathsf{srs\_g1}, \hat{f}, \mathbf{z})$:
\State $v \gets \mathsf{evaluate\_mle}(\hat{f}, \mathbf{z})$;\;
       $\mathsf{bases} \gets \mathsf{srs\_g1}.\mathsf{to\_affine}()$
\State $\mathsf{table} \gets \hat{f}$;\; $\mathsf{proofs} \gets []$
\For{$i = 0$ \textbf{to} $\mu - 1$}
    \For{$\mathsf{idx} = 0$ \textbf{to} $2^\mu - 1$}
        \State $j \gets (\mathsf{idx} \gg (i+1))$ \textbf{and} $(2^{\mu-i-1}-1)$
        \State $w[\mathsf{idx}] \gets \mathsf{table}[2j+1] - \mathsf{table}[2j]$
    \EndFor
    \State $\mathsf{proofs}.\mathsf{push}(\mathsf{MSM}(\mathsf{bases}, w))$
    \State $\mathsf{table}[j] \gets (1-z_i)\mathsf{table}[2j] + z_i\mathsf{table}[2j+1]$;\;
           halve table
\EndFor
\State \Return $(v,\; \mathsf{proofs})$
\Statex
\Statex \textbf{Verify}$(g_1, g_2, \mathsf{g2\_r}, C, \mathbf{z}, v, \mathsf{proofs})$:
\State $\mathsf{lhs} \gets e(C - v \cdot g_1,\; g_2)$;\;
       $\mathsf{rhs} \gets 0_{\mathbb{G}_T}$
\For{$i = 0$ \textbf{to} $\mu - 1$}
    \State $\mathsf{rhs} \mathrel{+}= e(\mathsf{proofs}[i],\;
           \mathsf{g2\_r}[i] - z_i \cdot g_2)$
\EndFor
\State \Return $\mathsf{lhs} = \mathsf{rhs}$
\end{algorithmic}
\end{algorithm}

\section{Implementation}

The implementation follows the same generic \texttt{Pairing} trait
approach in Rust using \texttt{arkworks} \cite{arkworks}. Commitment
uses Pippenger's MSM via \texttt{VariableBaseMSM}. The key
implementation detail is in witness computation: the SRS affine
points are converted \emph{once} before the outer loop and reused
across all $\mu$ MSMs, avoiding $\mu$ redundant affine conversions.
Verification uses \texttt{PairingOutput::zero()} as the additive
identity in $\mathbb{G}_T$ and accumulates pairing results with
\texttt{+=}.
\\\\

\section{Discussion and Summary}

The dominant cost in MKZG is proving, which scales exponentially
with $\mu$. At $\mu = 14$, proving takes over 5 seconds on BLS12-381.
This limits practical use to moderate $\mu$. However, verification
at 33 ms on BLS12-381 at $\mu = 14$ is a major strength compared to
Bulletproofs whose verification is $O(2^\mu)$. BN254 is the fastest
curve at roughly 2x across all metrics.
